前后端文件上传

本文针对前后端上传文件时的流程和关键点进行梳理。

const express = require("express");
const router = express.Router();
const multer = require("multer");           //1.用来处理enctype="multipart/form-data" 的请求
const fs = require("fs");
const path = require("path");
var upload = multer({ dest: 'upload/' });

//通过post传递的数据
router.post("/upload", upload.array('files', 10), (req, res) => {
   var files = req.files;
   var fileInfos = [];

   // 获取文件信息
   for (let i in files) {
      var file = files[i];
      var fileInfo = {};
			
      fileInfo.encoding = file.encoding;
      fileInfo.destination = file.destination;
      fileInfo.mimetype = file.mimetype;
      fileInfo.originalname = file.originalname;
      fileInfo.size = file.size;
      fileInfo.path = file.path;
      fileInfos.push(fileInfo);

      fs.renameSync(file.path, file.destination + file.originalname, (err) => {
         err ? res.send('上传失败:' + err) : res.send('上传成功');
      })
   }
   res.set({
      'content-type': 'application/json; charset=utf-8'
   });
   res.end(JSON.stringify(fileInfos), 'utf8');
})

module.exports = router;
